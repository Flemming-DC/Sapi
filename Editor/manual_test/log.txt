    JOIN tab00 USING (tab0_id))
SELECT 
    cte.col00_2,
    tab10.col10_2,
    (SELECT count(tab20.col20_2) FROM tab20)
--FROM tree
--join cte ON tree.col_1 = cte.col0_1
FROM cte 
join tab ON tab.col_1 = cte.col0_1
JOIN tab1 USING (tab_id)
